# common/msan-ignorelist.txt
# Compile-time MSan ignorelist for libc++ iostream false positives.
#
# The system libc++ is not MSan-instrumented, so internal stream/locale state
# appears uninitialized. The stream-unpoison header handles the stream objects
# themselves, but libc++ also reads locale facet state that we cannot easily
# unpoison. Suppress reports inside the standard stream output functions that
# doctest uses for reporting; this only affects test-framework output, not the
# hash map code under test.

[memory]

# char/char traits insertion used by doctest's console reporter
fun:*__put_character_sequence*

# Extra safety: doctest's version/summary printing functions live in the test
# TU and may inline the uninstrumented stream reads.
fun:*printVersion*
fun:*test_run_start*
fun:*test_run_end*
fun:*logTestStart*
fun:*logTestEnd*

# libc++ std::string / std::hash<std::string> read SSO padding and other
# uninstrumented internal state. These are standard-library false positives.
fun:*__string_hash*
fun:*_Hash_impl*
fun:*basic_string*__is_long*
fun:*basic_string*__get_pointer*
fun:*basic_string*data*
fun:*basic_string*size*
